\begin{multicols}{2}

\textbf{Author:} Da

\textbf{Source:} \href{https://discord.com/channels/740870068178649108/1161571991732625468}{Archon Studios Discord}

\textit{The ancient dragons have awoken from their septcentennial slumber, unleashing chaos upon Erathia. Age-old enemies must set aside centuries of conflict to confront the draconic threat, lest they see the continent consumed by fire.}  % no-check-caps

\subsection*{\MakeUppercase{Scenario Length}}
This Scenario plays out over 12 Rounds.

\subsection*{\MakeUppercase{Player Setup}}
\textbf{Player Count:} 4 players.

\textbf{Starting Resources:} 10 \svg{gold}, 2 \svg{building_materials}, 1 \svg{valuables}

\textbf{Starting Income:} 15 \svg{gold}, 2 \svg{building_materials}, 1 \svg{valuables}

\textbf{Starting Units:}

\begin{itemize}
  \item 2 × A Pack of cheapest \bronze\ Units.
  \item A Few chosen \bronze\ Units.
\end{itemize}

\textbf{Town Buildings:} \bronze\ Dwelling, Mage Guild \textit{Do not gain ``When Built'' bonus}.

\subsection*{\MakeUppercase{Map Setup}}
Take the following Map Tiles and arrange them as shown in the Scenario map layout:


\begin{itemize}
  \item 4 × Starting (I) Map Tiles.
  \item 8 × Far (II–III) Map Tiles.
  \item 8 × Near (IV–V) Map Tiles.
  \item 2 × Center (VI–VII) Map Tiles with Dragon Utopia.
\end{itemize}

\subsection*{\MakeUppercase{Victory Conditions}}
The game ends at the end of Round when either of these conditions are met:
\begin{itemize}
  \item Both Dragon Utopias have been flagged.
  \item One player has defeated each other player's Main Hero at least once.
\end{itemize}

\subsection*{\MakeUppercase{Defeat Conditions}}
If no Victory Condition has been met by the end of Round 12, all players lose.

\subsection*{\MakeUppercase{Victory Points}}
The player with the most Victory Points (VP) wins. In case of a tie, the player controlling more Dragon Utopias wins the game.
\begin{itemize}
  \item 4 VP for Flagging a Dragon Utopia
  \item 3 VP for each enemy Main Town captured \textit{once per Enemy}
  \item 2 VP for each enemy Main Hero defeated
\end{itemize}

\subsection*{\MakeUppercase{Timed Events}}
\textbf{\nth{1} Round:}
\begin{itemize}
  \item All Heroes gain +1 \svg{movement}
\end{itemize}
\textbf{\nth{6} Round:}
\begin{itemize}
  \item Remove all Far (II–III) map Tiles.
\end{itemize}
\textbf{\nth{9} Round:}
\begin{itemize}
  \item Remove all Near (IV–V) map Tiles.
\end{itemize}
\textbf{\nth{11} Round:}
\begin{itemize}
  \item Remove all Starting (I) map Tiles.
\end{itemize}

\subsection*{\MakeUppercase{Additional Rules}}
\begin{itemize}
  \item Players may not recruit Secondary Heroes.
  \item Players may use their M\&M Deck while paying to defend their Town.
  \item If a Hero stand on a removed Tile, move them to the nearest revealed Field and reduce their lowest initiative Unit's \svg{health_points} to 0. That player suffers -1 \svg{movement} this Round.
  \item Level VII Neutral Combats cannot be skipped.
  \item \textbf{Dragon Utopia:} Gain 10 \svg{gold}, 1 \svg{valuables} and Search (2) \svg{artifact} twice.
  \item \textbf{Obelisk:} Roll 1 \svg{resource} or \svg{treasure}.
  \item When Tiles are removed, reduce any production provided by mines or settlements on that Tile.
  \item Remove ``Astrologers proclaim week of the Magic Tortoise'' from the Astrologers Proclaim Deck.
\end{itemize}
